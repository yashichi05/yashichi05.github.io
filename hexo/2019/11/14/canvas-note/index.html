var ctx = canvas.getContext("2d");
var blockWidth = 200;
const PI = Math.PI;
const PI2 = Math.PI * 2;

canvas.width = blockWidth * 3; ///>???
canvas.height = blockWidth * 3;
ctx.fillCircle = function(x,y,r){
this.beginPath()
this.arc(x,y,r,0,PI2)
this.fill()

var color = {
red: "#F74456",
white: "#fff",
yellow: "#F1DA56",
blue: "#036FAF"
};

function drawBlock(pos, bg, draw, time) {
ctx.save();
ctx.translate(pos.x * blockWidth, pos.y * blockWidth);
ctx.fillStyle = bg;
ctx.fillRect(0, 0, blockWidth, blockWidth);//建立方塊
ctx.translate(100, 100);//座標移到方況的中間
draw(); //執行內部繪畫 function
ctx.restore();//復原座標
}
